## 2. Core Deployment Class

```typescript
// lib/tkoin-deployer.ts
import { 
  Connection, 
  Keypair, 
  PublicKey, 
  Transaction, 
  SystemProgram,
  sendAndConfirmTransaction 
} from '@solana/web3.js';
import { 
  createInitializeMintInstruction,
  createInitializeTransferFeeConfigInstruction,
  getMintLen,
  ExtensionType
} from '@solana/spl-token-2022';
import { 
  getMint,
  TOKEN_2022_PROGRAM_ID
} from '@solana/spl-token';
import { TOKEN_CONFIG } from '../config/token-config';

export class TkoinDeployer {
  private connection: Connection;
  private payer: Keypair;

  constructor(connection: Connection, payer: Keypair) {
    this.connection = connection;
    this.payer = payer;
  }

  /**
   * Deploys the TKOIN token with Token-2022 standard and transfer fee extension
   * @returns PublicKey of the deployed mint
   */
  async deployTkoinToken(): Promise<PublicKey> {
    console.log('Starting TKOIN Token-2022 deployment...');

    // Calculate required mint account size with transfer fee extension
    const extensions = [ExtensionType.TransferFeeConfig];
    const mintLen = getMintLen(extensions);
    const lamports = await this.connection.getMinimumBalanceForRentExemption(mintLen);

    // Generate mint keypair
    const mintKeypair = Keypair.generate();
    console.log(`Generated mint address: ${mintKeypair.publicKey.toString()}`);

    // Create mint account transaction
    const transaction = new Transaction();

    // Create account instruction
    const createAccountIx = SystemProgram.createAccount({
      fromPubkey: this.payer.publicKey,
      newAccountPubkey: mintKeypair.publicKey,
      space: mintLen,
      lamports,
      programId: TOKEN_2022_PROGRAM_ID,
    });

    // Initialize mint instruction
    const initializeMintIx = createInitializeMintInstruction(
      mintKeypair.publicKey,
      TOKEN_CONFIG.decimals,
      new PublicKey(TOKEN_CONFIG.mintAuthority),
      new PublicKey(TOKEN_CONFIG.mintAuthority),
      TOKEN_2022_PROGRAM_ID
    );

    // Initialize transfer fee config instruction
    const transferFeeConfigIx = createInitializeTransferFeeConfigInstruction(
      mintKeypair.publicKey,
      new PublicKey(TOKEN_CONFIG.transferFeeAuthority),
      new PublicKey(TOKEN_CONFIG.transferFeeAuthority),
      TOKEN_CONFIG.transferFeeBasisPoints,
      TOKEN_CONFIG.maxFeeBasisPoints,
      TOKEN_2022_PROGRAM_ID
    );

    // Add all instructions to transaction
    transaction.add(createAccountIx, initializeMintIx, transferFeeConfigIx);

    // Send and confirm transaction
    const signature = await sendAndConfirmTransaction(
      this.connection,
      transaction,
      [this.payer, mintKeypair],
      { commitment: 'confirmed' }
    );

    console.log(`TKOIN mint created successfully`);
    console.log(`Transaction: ${signature}`);
    console.log(`Mint Address: ${mintKeypair.publicKey.toString()}`);

    return mintKeypair.publicKey;
  }

  /**
   * Mints initial token supply to treasury wallet
   * @param mintAddress - The mint public key
   */
  async mintInitialSupply(mintAddress: PublicKey): Promise<void> {
    console.log(`Minting initial supply to treasury...`);

    const treasury = new PublicKey(TOKEN_CONFIG.treasuryWallet);
    const mintAmount = BigInt(TOKEN_CONFIG.initialSupply * Math.pow(10, TOKEN_CONFIG.decimals));

    // Note: In production, you would use createMintToInstruction from @solana/spl-token-2022
    // This is a placeholder for the actual minting logic
    console.log(`Initial supply allocation: ${TOKEN_CONFIG.initialSupply.toLocaleString()} TK`);
    console.log(`Treasury address: ${treasury.toString()}`);
    console.log(`Base units to mint: ${mintAmount.toString()}`);
  }

  /**
   * Verifies the token deployment was successful
   * @param mintAddress - The mint public key to verify
   */
  async verifyTokenDeployment(mintAddress: PublicKey): Promise<void> {
    console.log('Verifying token deployment...');
    
    try {
      const mintInfo = await getMint(
        this.connection,
        mintAddress,
        'confirmed',
        TOKEN_2022_PROGRAM_ID
      );

      console.log('Token verification successful:');
      console.log(`  Mint: ${mintAddress.toString()}`);
      console.log(`  Decimals: ${mintInfo.decimals}`);
      console.log(`  Supply: ${mintInfo.supply.toString()}`);
      console.log(`  Mint Authority: ${mintInfo.mintAuthority?.toString() || 'None'}`);
      console.log(`  Freeze Authority: ${mintInfo.freezeAuthority?.toString() || 'None'}`);

      // Verify extensions
      if (mintInfo.tlvData) {
        console.log(`  Extensions: TransferFeeConfig`);
      }

    } catch (error) {
      console.error('Token verification failed:', error);
      throw error;
    }
  }

  /**
   * Generates deployment report with all relevant information
   * @param mintAddress - The deployed mint public key
   */
  generateDeploymentReport(mintAddress: PublicKey): void {
    console.log('\n=== TKOIN DEPLOYMENT REPORT ===');
    console.log(`Token Name: ${TOKEN_CONFIG.name}`);
    console.log(`Symbol: ${TOKEN_CONFIG.symbol}`);
    console.log(`Mint Address: ${mintAddress.toString()}`);
    console.log(`Decimals: ${TOKEN_CONFIG.decimals}`);
    console.log(`Total Supply: ${TOKEN_CONFIG.totalSupply.toLocaleString()} ${TOKEN_CONFIG.symbol}`);
    console.log(`Initial Supply: ${TOKEN_CONFIG.initialSupply.toLocaleString()} ${TOKEN_CONFIG.symbol}`);
    console.log(`Transfer Fee: ${TOKEN_CONFIG.transferFeeBasisPoints / 100}%`);
    console.log(`Max Transfer Fee: ${TOKEN_CONFIG.maxFeeBasisPoints / 100}%`);
    console.log(`Treasury Wallet: ${TOKEN_CONFIG.treasuryWallet}`);
    console.log(`Token Standard: Token-2022`);
    console.log(`Logo URL: ${TOKEN_CONFIG.metadata.logoURI}`);
    console.log('================================');
  }
}
```

```typescript
// scripts/deploy-tkoin.ts
import { Connection, Keypair, clusterApiUrl } from '@solana/web3.js';
import { TkoinDeployer } from '../lib/tkoin-deployer';
import { TOKEN_CONFIG } from '../config/token-config';

/**
 * Main deployment function for TKOIN token
 */
async function deployTkoin() {
  console.log('TKOIN Token Deployment Initialized');
  console.log('===================================');

  // Initialize connection to Solana cluster
  const connection = new Connection(clusterApiUrl('devnet'), 'confirmed');
  
  // For mainnet deployment, use:
  // const connection = new Connection(clusterApiUrl('mainnet-beta'), 'confirmed');

  // Load payer keypair (in production, use secure key management)
  // This should be replaced with your actual deployment wallet
  const payer = Keypair.generate(); 
  console.log(`Deployer: ${payer.publicKey.toString()}`);

  const deployer = new TkoinDeployer(connection, payer);

  try {
    // Step 1: Deploy the TKOIN token
    console.log('\n--- Step 1: Deploying TKOIN Token ---');
    const mintAddress = await deployer.deployTkoinToken();

    // Step 2: Mint initial supply
    console.log('\n--- Step 2: Minting Initial Supply ---');
    await deployer.mintInitialSupply(mintAddress);

    // Step 3: Verify deployment
    console.log('\n--- Step 3: Verifying Deployment ---');
    await deployer.verifyTokenDeployment(mintAddress);

    // Step 4: Generate deployment report

    console.log('\nTKOIN DEPLOYMENT COMPLETED SUCCESSFULLY');

    console.log('1. Update configuration with actual treasury wallet');
    console.log('2. Deploy to mainnet when ready');
    console.log('3. Set up token distribution strategy');
    console.log('4. Integrate with Tkoin Protocol applications');

  } catch (error) {
    console.error('Deployment failed:', error);
    process.exit(1);
  }
}

// Execute deployment
deployTkoin().catch(console.error);
```

## 4. Utility Functions

```typescript
// lib/deployment-utils.ts
import { PublicKey, Connection } from '@solana/web3.js';
import { TOKEN_2022_PROGRAM_ID } from '@solana/spl-token';

/**
 * Utility functions for token deployment and management
 */
export class DeploymentUtils {
  /**
   * Validates a Solana public key
   */
  static validatePublicKey(address: string): boolean {
    try {
      new PublicKey(address);
      return true;
    } catch {
      return false;
    }
  }

  /**
   * Checks if a mint account exists and is valid
   */
  static async validateMintAccount(connection: Connection, mintAddress: PublicKey): Promise<boolean> {
    try {
      const accountInfo = await connection.getAccountInfo(mintAddress);
      return accountInfo !== null && accountInfo.owner.equals(TOKEN_2022_PROGRAM_ID);
    } catch {
      return false;
    }
  }

  /**
   * Generates explorer URLs for easy verification
   */
  static generateExplorerUrls(mintAddress: PublicKey, signature: string) {
    return {
      mint: `https://explorer.solana.com/address/${mintAddress.toString()}`,
      transaction: `https://explorer.solana.com/tx/${signature}`
    };
  }
}
```
